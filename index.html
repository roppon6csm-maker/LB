// RegEx parser stripping full shared profile URLs down to raw screenname handles function extractTwitterId(rawStr) { if (!rawStr) return ''; let src = rawStr.trim(); // https://x.com/ユーザー名?パラメータ の形式からユーザー名を抽出 if (src.includes('x.com/') || src.includes('twitter.com/')) { const parts = src.split(/[?#]/)[0].split('/'); return parts[parts.length - 1] || ''; } return src.replace(/@/g, ''); }
